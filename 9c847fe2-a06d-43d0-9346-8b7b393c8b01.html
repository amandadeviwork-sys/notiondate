<head>
  <meta charset="utf-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <title>Live Date & Time (English)</title>
  <style>
    :root { color-scheme: light dark; }
    body {
      margin: 0; padding: 0.25rem 0.5rem;
      font-family: system-ui, -apple-system, Segoe UI, Roboto, Helvetica, Arial, "Apple Color Emoji", "Segoe UI Emoji";
      font-size: 14px; line-height: 1.4;
      background: transparent; color: inherit;
      display: inline-block; white-space: nowrap;
    }
  </style>
</head>
<body>
  <time id="now" aria-label="Current date and time"></time>
  <script>
    const el = document.getElementById('now');

    // Fixed English output
    const locale = 'en-US';
    // Auto-detect each viewer’s time zone
    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;

    const fmtDate = new Intl.DateTimeFormat(locale, { dateStyle: 'long', timeZone: tz });
    const fmtTime = new Intl.DateTimeFormat(locale, {
      hour: '2-digit',
      minute: '2-digit',
      timeZone: tz
      // For 24-hour time, add: hourCycle: 'h23'
    });

    function render() {
      const d = new Date();
      el.textContent = `${fmtDate.format(d)} • ${fmtTime.format(d)}`;
      el.dateTime = d.toISOString();
    }

    render();
    setInterval(render, 1000); // keep it live
  </script>
</body>
